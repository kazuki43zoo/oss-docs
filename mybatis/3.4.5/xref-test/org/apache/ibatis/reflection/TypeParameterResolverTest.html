375   @Test
376   public void testReturnParam_WildcardWithUpperBounds() throws Exception {
377     class Key {}
378     @SuppressWarnings("unused")
379     class KeyBean<S extends Key & Cloneable, T extends Key> {
380       private S key1;
381       private T key2;
382       public S getKey1() {
383         return key1;
384       }
385       public void setKey1(S key1) {
386         this.key1 = key1;
387       }
388       public T getKey2() {
389         return key2;
390       }
391       public void setKey2(T key2) {
392         this.key2 = key2;
393       }
394     }
395     Class<?> clazz = KeyBean.class;
396     Method getter1 = clazz.getMethod("getKey1");
397     assertEquals(Key.class, TypeParameterResolver.resolveReturnType(getter1, clazz));
398     Method setter1 = clazz.getMethod("setKey1", Key.class);
399     assertEquals(Key.class, TypeParameterResolver.resolveParamTypes(setter1, clazz)[0]);
400     Method getter2 = clazz.getMethod("getKey2");
401     assertEquals(Key.class, TypeParameterResolver.resolveReturnType(getter2, clazz));
402     Method setter2 = clazz.getMethod("setKey2", Key.class);
403     assertEquals(Key.class, TypeParameterResolver.resolveParamTypes(setter2, clazz)[0]);
404   }
