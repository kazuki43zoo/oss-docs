44  /*
45   * This class contains tests for multiple results.  
46   * It is based on Jeff's ref cursor tests.
47   */
48  @Category(EmbeddedPostgresqlTests.class)
49  public class MultipleResultTest {
50  
51    private static final EmbeddedPostgres postgres = new EmbeddedPostgres();
52  
53    private static SqlSessionFactory sqlSessionFactory;
54  
55    @BeforeClass
56    public static void setUp() throws Exception {
57      // Launch PostgreSQL server. Download / unarchive if necessary.
58      String url = postgres.start(EmbeddedPostgres.cachedRuntimeConfig(Paths.get(System.getProperty("java.io.tmpdir"), "pgembed")), "localhost", SocketUtil.findFreePort(), "multiple_resultsets", "postgres", "root", Collections.emptyList());
59  
60      Configuration configuration = new Configuration();
61      Environment environment = new Environment("development", new JdbcTransactionFactory(), new UnpooledDataSource(
62          "org.postgresql.Driver", url, null));
63      configuration.setEnvironment(environment);
64      configuration.setMapUnderscoreToCamelCase(true);
65      configuration.addMapper(Mapper.class);
66      sqlSessionFactory = new SqlSessionFactoryBuilder().build(configuration);
67  
68      try (SqlSession session = sqlSessionFactory.openSession();
69          Connection conn = session.getConnection();
70          Reader reader = Resources.getResourceAsReader("org/apache/ibatis/submitted/usesjava8/multiple_resultsets/CreateDB.sql")) {
71        ScriptRunner runner = new ScriptRunner(conn);
72        runner.setLogWriter(null);
73        runner.runScript(reader);
74      }
75    }
76  
77    @AfterClass
78    public static void tearDown() {
79      postgres.stop();
80    }
81  
82    @Test
83    public void shouldGetMultipleResultSetsWithOneStatement() throws IOException {
84      try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
85        Mapper mapper = sqlSession.getMapper(Mapper.class);
86        List<List<?>> results = mapper.getUsersAndGroups();
87        Assert.assertEquals(2, results.size());
88  
89        Assert.assertEquals(6, results.get(0).size());
90        OrderDetail detail = (OrderDetail)results.get(0).get(0);
91        Assert.assertEquals(1, detail.getOrderId());
92        Assert.assertEquals(1, detail.getLineNumber());
93        Assert.assertEquals(1, detail.getQuantity());
94        Assert.assertEquals("Pen", detail.getItemDescription());
95  
96        Assert.assertEquals(2, results.get(1).size());
97        OrderHeader header = (OrderHeader)results.get(1).get(0);
98        Assert.assertEquals(1, header.getOrderId());
99        Assert.assertEquals("Fred", header.getCustName());
100     }
101   }
102 
103   @Test
104   public void shouldSkipNullResultSet() {
105     try (SqlSession sqlSession = sqlSessionFactory.openSession()) {
106       try {
107         Mapper mapper = sqlSession.getMapper(Mapper.class);
108         List<List<?>> results = mapper.multiResultsWithUpdate();
109         Assert.assertEquals(2, results.size());
110 
111         Assert.assertEquals(6, results.get(0).size());
112         OrderDetail detail = (OrderDetail) results.get(0).get(0);
113         Assert.assertEquals(1, detail.getOrderId());
114         Assert.assertEquals(1, detail.getLineNumber());
115         Assert.assertEquals(1, detail.getQuantity());
116         Assert.assertEquals("Pen", detail.getItemDescription());
117 
118         Assert.assertEquals(2, results.get(1).size());
119         OrderHeader header = (OrderHeader) results.get(1).get(0);
120         Assert.assertEquals(1, header.getOrderId());
121         Assert.assertEquals("Fred", header.getCustName());
122 
123         results = mapper.getUsersAndGroups();
124         Assert.assertEquals(7, results.get(0).size());
125         detail = (OrderDetail) results.get(0).get(6);
126         Assert.assertEquals(2, detail.getOrderId());
127         Assert.assertEquals(4, detail.getLineNumber());
128         Assert.assertEquals(5, detail.getQuantity());
129         Assert.assertEquals("Eraser", detail.getItemDescription());
130       } finally {
131         sqlSession.rollback(true);
132       }
133     }
134   }
