159     assertSame(factory.getConfiguration().getDefaultExecutorType(), ExecutorType.REUSE);

246   @Test
247   public void testSetConfigLocation() throws Exception {
248     setupFactoryBean();
249 
250     factoryBean.setConfigLocation(new org.springframework.core.io.ClassPathResource(
251         "org/mybatis/spring/mybatis-config.xml"));
252 
253     SqlSessionFactory factory = factoryBean.getObject();
254 
255     assertEquals(factory.getConfiguration().getEnvironment().getId(), SqlSessionFactoryBean.class.getSimpleName());
256     assertSame(factory.getConfiguration().getEnvironment().getDataSource(), dataSource);
257     assertSame(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass(),
258         org.mybatis.spring.transaction.SpringManagedTransactionFactory.class);
259     assertSame(factory.getConfiguration().getVfsImpl(), JBoss6VFS.class);
260 
261     // properties explicitly set differently than the defaults in the config xml
262     assertFalse(factory.getConfiguration().isCacheEnabled());
263     assertTrue(factory.getConfiguration().isUseGeneratedKeys());
264     assertSame(factory.getConfiguration().getDefaultExecutorType(), org.apache.ibatis.session.ExecutorType.REUSE);
265 
266     // for each statement in the xml file: org.mybatis.spring.TestMapper.xxx & xxx
267     assertEquals(8, factory.getConfiguration().getMappedStatementNames().size());
268 
269     assertEquals(0, factory.getConfiguration().getResultMapNames().size());
270     assertEquals(0, factory.getConfiguration().getParameterMapNames().size());
271   }

418   private void assertDefaultConfig(SqlSessionFactory factory) {
419     assertConfig(factory, SqlSessionFactoryBean.class.getSimpleName(),
420         org.mybatis.spring.transaction.SpringManagedTransactionFactory.class);
421     assertEquals(0, factory.getConfiguration().getVariables().size());
422   }
423 
424   private void assertConfig(SqlSessionFactory factory, Class<? extends TransactionFactory> transactionFactoryClass) {
425     assertConfig(factory, SqlSessionFactoryBean.class.getSimpleName(), transactionFactoryClass);
426   }
427 
428   private void assertConfig(SqlSessionFactory factory, String environment,
429       Class<? extends TransactionFactory> transactionFactoryClass) {
430     assertEquals(factory.getConfiguration().getEnvironment().getId(), environment);
431     assertSame(factory.getConfiguration().getEnvironment().getDataSource(), dataSource);
432     assertSame(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass(),
433         transactionFactoryClass);
434 
435     // no mappers configured => no mapped statements or other parsed elements
436     assertEquals(factory.getConfiguration().getMappedStatementNames().size(), 0);
437     assertEquals(factory.getConfiguration().getResultMapNames().size(), 0);
438     assertEquals(factory.getConfiguration().getParameterMapNames().size(), 0);
439     assertEquals(factory.getConfiguration().getSqlFragments().size(), 0);
440   }
